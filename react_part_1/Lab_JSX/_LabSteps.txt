<Note>完整的 <script> 內容如下:

        const dataItem = {

            image: "images/rileyi.jpg"
        }

        function formatName(species) {
            return <span>
                {species.commonName}<br />
                {species.scientificName}
            </span>;
        }

        const element = (
            <div className="col-4">
                <h3> {formatName(dataItem)} </h3>
                <img src={dataItem.image} />
            </div>
        );

        ReactDOM.render(
            element,
            document.getElementById('root')
        );


    </script>
